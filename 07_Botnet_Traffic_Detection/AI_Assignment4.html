class DNNModel(nn.Module):
    def __init__(self, input_dim, hidden_dims, num_classes, dropout_rate=0.3):
        super().__init__()
        layers = []
        
        # Input layer
        layers.append(nn.Linear(input_dim, hidden_dims[0]))
        layers.append(nn.ReLU())
        layers.append(nn.Dropout(dropout_rate))
        
        # Hidden layers
        for i in range(len(hidden_dims)-1):
            layers.append(nn.Linear(hidden_dims[i], hidden_dims[i+1]))
            layers.append(nn.ReLU())
            layers.append(nn.Dropout(dropout_rate))
        
        # Output layer
        layers.append(nn.Linear(hidden_dims[-1], num_classes))
        
        self.network = nn.Sequential(*layers)
    
    def forward(self, x):
        return self.network(x)

    def print_model_summary(self):
        """
        Print a summary of the model architecture
        """
        print("Deep Neural Network Architecture:")
        print("-" * 40)
        for name, module in self.network.named_children():
            if isinstance(module, nn.Linear):
                print(f"Layer: Linear - Input: {module.in_features}, Output: {module.out_features}")
            elif isinstance(module, nn.ReLU):
                print("Activation: ReLU")
            elif isinstance(module, nn.Dropout):
                print(f"Dropout: Rate {module.p}")
        print("-" * 40)

    def plot_model_architecture(self, metrics=None, dropout_rates=None):
        """
        Create an advanced visualization of the neural network architecture
        """
        plt.figure(figsize=(15, 8))

        # Network architecture details
        layer_sizes = [6, 128, 64, 32, 1]
        layer_names = ['Input', 'Hidden Layer 1', 'Hidden Layer 2', 'Hidden Layer 3', 'Output']

        # Create a graph
        G = nx.DiGraph()

        # Position for layers
        pos = {}
        v_spacing = 1
        h_spacing = 1

        # Create nodes and calculate positions
        for layer, (size, name) in enumerate(zip(layer_sizes, layer_names)):
            for neuron in range(size):
                # Spread neurons vertically
                y = (size/2 - neuron) * v_spacing
                x = layer * h_spacing * 3  # Wider spacing between layers
                node = f"{name}\nNeuron {neuron+1}"
                G.add_node(node)
                pos[node] = (x, y)

        # Connect layers
        for layer in range(len(layer_sizes)-1):
            source_nodes = [n for n in G.nodes() if layer_names[layer] in n]
            target_nodes = [n for n in G.nodes() if layer_names[layer+1] in n]
    
            for src in source_nodes:
                for tgt in target_nodes:
                    G.add_edge(src, tgt)

        # Draw the graph
        plt.subplot(121)  # Left subplot for network structure
        nx.draw(G, pos, 
                with_labels=True, 
                node_color='lightblue', 
                node_size=300, 
                font_size=6, 
                font_weight='bold',
                arrows=True,
                edge_color='gray',
                alpha=0.7)
        plt.title('Neural Network Architecture', fontsize=10)

        # Performance subplot
        if metrics and dropout_rates:
            plt.subplot(122)
            plt.bar(dropout_rates, metrics, color=['blue', 'green', 'red'])
            plt.title('Model Performance by Dropout Rate', fontsize=10)
            plt.xlabel('Dropout Rate')
            plt.ylabel('F1 Score')
            for i, v in enumerate(metrics):
                plt.text(dropout_rates[i], v, f'{v:.3f}', ha='center', va='bottom')
        plt.tight_layout()
        plt.show()    
        
    

class BotnetDetectionPipeline:
    def __init__(self):
        self.label_encoder = LabelEncoder()
        self.scaler = StandardScaler()
        self.models = {
            'svc_binary': None,
            'svc_type': None,
            'dnn_binary': None,
            'dnn_type': None
        }
        self.results_table = pd.DataFrame()
        self.device = torch.device('cuda' if torch.cuda.is_available() else 'cpu')
        print(f"Using device: {self.device}")

    def create_results_table(self):
        """Initialize results tracking table."""
        columns = {
            'Model': [
                'Kernel SVC for botnet vs normal',
                'Kernel SVC for type of botnet',
                'DNN for botnet vs normal',
                'DNN for type of botnet'
            ],
            'Config 1 F1': [0.0] * 4,
            'Config 2 F1': [0.0] * 4,
            'Config 3 F1': [0.0] * 4,
            'Best Config': [''] * 4,
            'Best Train Time (s)': [0.0] * 4,
            'Best Test F1': [0.0] * 4,
            'Interpretability Status': ['Not Done'] * 4
        }
    
        # Create DataFrame from the columns dictionary
        self.results_table = pd.DataFrame(columns)
    
        return self.results_table
        

    def load_and_preprocess_data(self, base_path, sample_size=1000, random_state=42):
        """Load and preprocess the CTU-13 dataset with sampling."""
        print("Loading and preprocessing data...")
        
        # List all files in the directory
        all_files = []
        for root, dirs, files in os.walk(base_path):
            for file in files:
                if file.endswith('.binetflow'):
                    all_files.append(os.path.join(root, file))
        
        print(f"Found {len(all_files)} files")
        
        dfs = []
        for file in all_files:
            try:
                df = pd.read_csv(file)
                file_sample_size = sample_size // len(all_files)
                if len(df) > file_sample_size:
                    df = df.sample(n=file_sample_size, random_state=random_state)
                scenario = os.path.basename(os.path.dirname(file))
                df['Scenario'] = scenario
                dfs.append(df)
            except Exception as e:
                print(f"Error reading file {file}: {e}")
        
        combined_df = pd.concat(dfs, ignore_index=True)
        
        # Create binary and type labels
        combined_df['is_botnet'] = combined_df['Label'].apply(
            lambda x: 1 if 'Botnet' in x or 'malicious' in x.lower() else 0
        )
        
        def get_botnet_type(label):
            if 'Botnet' not in label and 'malicious' not in label.lower():
                return 'Normal'
            if 'ZeroAccess' in label:
                return 'ZeroAccess'
            elif 'Kaiten' in label:
                return 'Kaiten'
            elif 'Mirai' in label:
                return 'Mirai'
            else:
                return 'Other'
        
        combined_df['botnet_type'] = combined_df['Label'].apply(get_botnet_type)
        
        # Prepare numeric features
        numeric_features = ['Dur', 'Proto', 'SrcBytes', 'TotBytes', 'State', 'TotPkts']
        
        # Convert categorical features to numeric
        proto_mapping = {proto: idx for idx, proto in 
                        enumerate(combined_df['Proto'].unique())}
        state_mapping = {state: idx for idx, state in 
                        enumerate(combined_df['State'].unique())}
        
        combined_df['Proto'] = combined_df['Proto'].map(proto_mapping)
        combined_df['State'] = combined_df['State'].map(state_mapping)
        
        # Prepare features and labels
        X = combined_df[numeric_features].copy()
        X = self.scaler.fit_transform(X)
        y_binary = combined_df['is_botnet']
        y_type = self.label_encoder.fit_transform(combined_df['botnet_type'])
        
        # Sequential processing flag
        needs_type_classification = combined_df['is_botnet'] == 1
        
        return X, y_binary, y_type, needs_type_classification

    def train_svc_model(self, X_train, y_train, config):
        """Train SVC model with given configuration."""
        model = SVC(**config, probability=True, random_state=42)
        start_time = time.time()
        model.fit(X_train, y_train)
        train_time = time.time() - start_time
        return model, train_time

    def train_dnn_model(self, X_train, y_train, input_shape, num_classes, config):
        """Train DNN model with PyTorch."""
        X_tensor = torch.FloatTensor(X_train).to(self.device)
        y_tensor = torch.LongTensor(y_train).to(self.device)
    
        dataset = TensorDataset(X_tensor, y_tensor)
        train_loader = DataLoader(dataset, batch_size=config['batch_size'], shuffle=True)
    
        model = DNNModel(
            input_dim=input_shape,
            hidden_dims=[128, 64, 32],
            num_classes=1 if num_classes == 2 else num_classes,  # Change output dimension for binary case
            dropout_rate=config['dropout_rate']
        ).to(self.device)
    
        # Use BCEWithLogitsLoss for binary classification
        criterion = nn.BCEWithLogitsLoss() if num_classes == 2 else nn.CrossEntropyLoss()
        optimizer = optim.Adam(model.parameters(), lr=0.001)
        scheduler = optim.lr_scheduler.ReduceLROnPlateau(optimizer, mode='min', factor=0.5, patience=3)

        start_time = time.time()
        best_loss = float('inf')
        patience = 7
        patience_counter = 0
    
        for epoch in range(config['epochs']):
            model.train()
            total_loss = 0
            for batch_X, batch_y in train_loader:
                optimizer.zero_grad()
                outputs = model(batch_X)
            
                if num_classes == 2:
                    outputs = outputs.squeeze()  # Ensure correct shape for binary classification
                    batch_y = batch_y.float()   # Convert targets to float for BCEWithLogitsLoss
            
                loss = criterion(outputs, batch_y)
                loss.backward()
                optimizer.step()
                total_loss += loss.item()
        
            avg_loss = total_loss / len(train_loader)
            if avg_loss < best_loss:
                best_loss = avg_loss
                patience_counter = 0
            else:
                patience_counter += 1
                if patience_counter >= patience:
                    print(f"Early stopping at epoch {epoch}")
                    break
    
        train_time = time.time() - start_time
        return model, train_time

    def evaluate_dnn_model(self, model, X_test, y_test, num_classes):
        """Evaluate DNN model."""
        model.eval()
        with torch.no_grad():
            X_tensor = torch.FloatTensor(X_test).to(self.device)
            outputs = model(X_tensor)
        
            if num_classes == 2:
                outputs = outputs.squeeze()
                predictions = (torch.sigmoid(outputs) > 0.5).cpu().numpy().astype(int)
            else:
                predictions = outputs.argmax(dim=1).cpu().numpy()
            
        return predictions

    def evaluate_model(self, model, X_test, y_test, is_binary=True):
        """Evaluate model performance."""
        if isinstance(model, nn.Module):
            y_pred = self.evaluate_dnn_model(
                model, X_test, y_test, 
                num_classes=2 if is_binary else len(np.unique(y_test))
            )
        else:
            y_pred = model.predict(X_test)
            
        cm = confusion_matrix(y_test, y_pred)
        f1 = f1_score(y_test, y_pred, average='weighted')
        return {'confusion_matrix': cm, 'f1_score': f1}

    

    def plot_confusion_matrix(self, cm, model_name, is_binary=True):
        """Plot the confusion matrix."""
        plt.figure(figsize=(6, 6))
        labels = ['Normal', 'Botnet'] if is_binary else ['Normal', 'ZeroAccess', 'Kaiten', 'Mirai', 'Other']
        sns.heatmap(cm, annot=True, fmt="d", cmap="Blues", xticklabels=labels, yticklabels=labels)
        plt.title(f'Confusion Matrix - {model_name}')
        plt.ylabel('True label')
        plt.xlabel('Predicted label')
        plt.show()

    def interpret_model(self, model, X_train, model_type):
        """Generate model interpretability analysis."""
        if isinstance(model, nn.Module):
            # For PyTorch DNN
            def predict_fn(x):
                model.eval()
                with torch.no_grad():
                    return model(torch.FloatTensor(x).to(self.device)).cpu().numpy()
            explainer = shap.KernelExplainer(predict_fn, X_train[:100])
        else:
            # For SVC
            explainer = shap.KernelExplainer(model.predict_proba, X_train[:100])
            
        shap_values = explainer.shap_values(X_train[:10])
        
        plt.figure(figsize=(10, 6))
        if isinstance(shap_values, list):
            shap.summary_plot(shap_values[1], X_train[:10], show=False)
        else:
            shap.summary_plot(shap_values, X_train[:10], show=False)
        plt.title(f'SHAP Values for {model_type}')
        plt.tight_layout()
        plt.show()

    def train_and_evaluate_models(self, X_train, X_test, y_train_binary, y_test_binary):
        """Train and evaluate all model configurations."""
        print("\nTraining binary classification models...")
    
        svc_configs = [
            {'kernel': 'rbf', 'C': 1.0},
            {'kernel': 'rbf', 'C': 10.0},
            {'kernel': 'linear', 'C': 1.0}
        ]
    
        dnn_configs = [
            {'dropout_rate': 0.3, 'epochs': 50, 'batch_size': 32},
            {'dropout_rate': 0.5, 'epochs': 100, 'batch_size': 64},
            {'dropout_rate': 0.2, 'epochs': 75, 'batch_size': 48}
        ]

        for model_name, configs in [('svc_binary', svc_configs), ('dnn_binary', dnn_configs)]:
            best_f1 = 0
            best_train_time = float('inf')
            best_model = None
            # Store metrics and dropout rates for visualization
            metrics = []
            dropout_rates = []
        
            for i, config in enumerate(configs, 1):
                try:
                    print(f"\nTraining {model_name} with config {i}: {config}")
                    if 'svc' in model_name:
                        model, train_time = self.train_svc_model(X_train, y_train_binary, config)
                    else:
                        model, train_time = self.train_dnn_model(
                            X_train, y_train_binary, X_train.shape[1], 2, config
                        )
                
                    evaluation = self.evaluate_model(model, X_test, y_test_binary)
                    print(f"F1 Score: {evaluation['f1_score']:.3f}")

                    # Collect metrics for visualization
                    metrics.append(evaluation['f1_score'])
                    dropout_rates.append(config.get('dropout_rate', 0))
                
                    # Plot confusion matrix for each configuration
                    self.plot_confusion_matrix(
                        evaluation['confusion_matrix'],
                        f"{model_name} - Config {i}",
                        is_binary=True
                    )
                
                    # Update results table
                    model_type = 'Kernel SVC' if 'svc' in model_name else 'DNN'
                    idx = self.results_table.index[
                        self.results_table['Model'] == f'{model_type} for botnet vs normal'
                    ].item()
                    self.results_table.at[idx, f'Config {i} F1'] = evaluation['f1_score']
                
                    # Track best model
                    if evaluation['f1_score'] > best_f1:
                        best_f1 = evaluation['f1_score']
                        best_train_time = train_time
                        best_model = model
                        best_config_idx = i
            
                except Exception as e:
                    print(f"Error training {model_name} with config {i}: {str(e)}")
                    model_type = 'Kernel SVC' if 'svc' in model_name else 'DNN'
                    idx = self.results_table.index[
                        self.results_table['Model'] == f'{model_type} for botnet vs normal'
                    ].item()
                    self.results_table.at[idx, f'Issues with Config {i}'] = f"Error: {str(e)}"
        
            # Visualization of model architecture and performance for DNN
            if 'dnn' in model_name and best_model is not None:
                best_model.plot_model_architecture(metrics, dropout_rates)
        
            # Store best model and update results
            if best_model is not None:
                self.models[model_name] = best_model
                idx = self.results_table.index[
                    self.results_table['Model'] == f'{model_type} for botnet vs normal'
                ].item()
                self.results_table.at[idx, 'Best Config'] = f'Config {best_config_idx}'
                self.results_table.at[idx, 'Best Train Time (s)'] = best_train_time
                self.results_table.at[idx, 'Best Test F1'] = best_f1
            
                # Generate interpretability analysis for best model
                try:
                    self.interpret_model(best_model, X_train, model_type)
                    self.results_table.at[idx, 'Interpretability Status'] = 'Done'
                except Exception as e:
                    print(f"Error generating interpretability analysis: {str(e)}")
                    self.results_table.at[idx, 'Interpretability Status'] = f'Error: {str(e)}'



def main():
    # Initialize pipeline
    pipeline = BotnetDetectionPipeline()
    results_table = pipeline.create_results_table()
    print("Initial Results Table:")
    print(results_table)
    
    # Load data
    base_path = r"C:\Users\heebs\Downloads\CTU-13-Dataset\CTU-13-Dataset"
    X, y_binary, y_type, needs_type_classification = pipeline.load_and_preprocess_data(
        base_path,
        sample_size=2000
    )
    
    # Split data for binary classification
    X_train, X_test, y_train_binary, y_test_binary = train_test_split(
        X, y_binary, test_size=0.2, random_state=42
    )
    
    # Train and evaluate models
    pipeline.train_and_evaluate_models(X_train, X_test, y_train_binary, y_test_binary)
    
    # Print final results
    print("\nFinal Results Table:")
    pd.set_option('display.max_columns', None)
    pd.set_option('display.width', None)
    print(pipeline.results_table.to_string())

if __name__ == "__main__":
    main()
